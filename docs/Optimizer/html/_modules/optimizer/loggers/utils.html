        """Overridden default method.
        Deals with converting unserializable numeric types to builtin types.

        Args:
            o: Any object.

        Returns:
            Any.
        """
        if isinstance(o, np.generic) and np.isreal(o):
            return o.item()

        return super(NumpyEncoder, self).default(o)
